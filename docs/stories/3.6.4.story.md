As a **platform operator**, I want to **view real-time cost metrics comparing database storage vs. S3 archival savings**, so that **I can justify infrastructure spend and track ROI of archival strategy**.

## Technical Notes

### Database Schema

```sql
CREATE TABLE cost_snapshots (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  timestamp TIMESTAMP DEFAULT now(),
  db_storage_gb DECIMAL(10,2) NOT NULL,
  s3_storage_gb DECIMAL(10,2) NOT NULL,
  db_cost_monthly DECIMAL(10,4) NOT NULL, -- RDS: $1.5/GB/month
  s3_cost_monthly DECIMAL(10,4) NOT NULL, -- S3: $0.023/GB/month
  monthly_savings DECIMAL(10,4) NOT NULL, -- (db_cost - s3_cost)
  seven_year_savings DECIMAL(12,4) NOT NULL, -- 7 * 12 * monthly_savings
  compression_ratio DECIMAL(5,2) NOT NULL, -- e.g., 3.5
  accuracy_percent DECIMAL(5,2) NOT NULL, -- ±5% (calculated vs actual)

  UNIQUE(user_id, timestamp) -- One snapshot per user per day
);

-- RLS Policy: Users can only see own cost snapshots
ALTER TABLE cost_snapshots ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can only see own cost snapshots"
  ON cost_snapshots FOR SELECT
  USING (auth.uid() = user_id);

-- Index for dashboard queries
CREATE INDEX idx_cost_snapshots_user_time
  ON cost_snapshots(user_id, timestamp DESC);
```

### Cost Model & Formulas

```typescript
const COST_MODEL = {
  rds: {
    pricePerGb: 1.5, // $/GB/month (AWS RDS gp2)
    unit: "GB"
  },
  s3: {
    pricePerGb: 0.023, // $/GB/month (S3 Standard, first 1TB tier)
    compressionRatio: 3.5 // typical gzip for JSON/text execution logs
  }
};

// Cost Calculation
function calculateCostMetrics(dbStorageGb: number, s3StorageGb: number) {
  const dbCostMonthly = dbStorageGb * COST_MODEL.rds.pricePerGb;
  const s3CostMonthly = s3StorageGb * COST_MODEL.s3.pricePerGb;
  const savings = dbCostMonthly - s3CostMonthly;
  const sevenYearSavings = savings * 7 * 12;

  return {
    dbCost: dbCostMonthly,
    s3Cost: s3CostMonthly,
    monthlySavings: savings,
    sevenYearProjection: sevenYearSavings,
    compressionRatio: COST_MODEL.s3.compressionRatio,
    accuracy: 95 // ±5% means 95-105% accuracy band
  };
}

// Storage Growth Trend (7-day moving average)
function calculateStorageGrowthTrend(snapshots: CostSnapshot[]): TrendPoint[] {
  const last7Days = snapshots.slice(-7);
  return last7Days.map((snapshot, idx) => ({
    date: snapshot.timestamp,
    archivalRateGbPerDay: (snapshot.s3_storage_gb - (last7Days[idx-1]?.s3_storage_gb ?? 0))
  }));
}
```

### Nightly Job Implementation (node-schedule)

```typescript
// Schedule cost snapshot job
schedule.scheduleJob('0 2 * * *', async () => { // 2:00 AM UTC daily
  try {
    console.log('[CostJob] Starting nightly cost snapshot...');

    for (const userId of await getAllActiveUsers()) {
      try {
        // 1. Query DB storage for this user
        const dbStorage = await calculateUserDbStorage(userId);

        // 2. Query S3 archival size
        const s3Storage = await calculateUserS3Storage(userId);

        // 3. Reuse S3ArchivalService to get cost breakdown
        const costSavings = await S3ArchivalService.calculateCostSavings({
          userId,
          dbStorageGb: dbStorage,
          s3StorageGb: s3Storage
        });

        // 4. Persist snapshot
        await insertCostSnapshot({
          user_id: userId,
          db_storage_gb: dbStorage,
          s3_storage_gb: s3Storage,
          db_cost_monthly: costSavings.dbCost,
          s3_cost_monthly: costSavings.s3Cost,
          monthly_savings: costSavings.savings,
          seven_year_savings: costSavings.sevenYearProjection,
          compression_ratio: costSavings.compressionRatio,
          accuracy_percent: 95
        });

        console.log(`[CostJob] Snapshot for user ${userId}: saved $${costSavings.savings.toFixed(2)}/month`);
      } catch (error) {
        console.error(`[CostJob] Error processing user ${userId}:`, error);
        // Continue with next user, don't fail entire job
      }
    }

    console.log('[CostJob] Cost snapshot complete');
  } catch (error) {
    console.error('[CostJob] Fatal error:', error);
    // Alert ops: Send to Slack/monitoring
    await alertOps('Cost snapshot job failed', error);
  }
});
```

### Frontend Cost Card Component

```typescript
// In DashboardContainer (Story 3.6.2)
<CostCard
  dbCost={metrics.dbCost}
  s3Cost={metrics.s3Cost}
  monthlySavings={metrics.monthlySavings}
  sevenYearProjection={metrics.sevenYearProjection}
  storageGrowthTrend={metrics.storageGrowthTrend}
  trend={metrics.monthlySavings >= previousMonth ? 'up' : 'down'}
  trendColor={metrics.monthlySavings > 0 ? 'green' : 'red'}
/>

// CostCard Component
interface CostCardProps {
  dbCost: number;
  s3Cost: number;
  monthlySavings: number;
  sevenYearProjection: number;
  storageGrowthTrend: TrendPoint[];
  trend: 'up' | 'down' | 'stable';
  trendColor: 'green' | 'red' | 'gray';
}

export const CostCard: React.FC<CostCardProps> = ({
  dbCost,
  s3Cost,
  monthlySavings,
  sevenYearProjection,
  storageGrowthTrend,
  trend,
  trendColor
}) => (
  <div className="cost-card">
    <h3>Cost Savings from S3 Archival</h3>

    <div className="metrics-grid">
      <Metric label="DB Storage Cost" value={`$${dbCost.toFixed(2)}/mo`} />
      <Metric label="S3 Archival Cost" value={`$${s3Cost.toFixed(2)}/mo`} />
      <Metric label="Monthly Savings" value={`$${monthlySavings.toFixed(2)}`} highlight={true} />
      <Metric label="7-Year Projection" value={`$${sevenYearProjection.toFixed(0)}`} />
    </div>

    {/* Trend Chart */}
    <LineChart data={storageGrowthTrend} height={150}>
      <Line type="monotone" dataKey="archivalRateGbPerDay" stroke="#059669" name="Archival Rate (GB/day)" />
    </LineChart>

    {/* Trend Indicator */}
    <div className={`trend-indicator ${trendColor}`}>
      {trend === 'up' && '↑'} {trend === 'down' && '↓'} {trend === 'stable' && '↔'}
      <span className="trend-label">
        {trend === 'up' && 'Increasing savings'}
        {trend === 'down' && 'Decreasing savings'}
        {trend === 'stable' && 'Stable savings'}
      </span>
    </div>

    {/* Tooltip */}
    <Tooltip>
      <p>DB: {dbCost.toFixed(2)} (RDS $1.5/GB/mo)</p>
      <p>S3: {s3Cost.toFixed(2)} (S3 $0.023/GB/mo)</p>
      <p>Compression: ~3.5x</p>
    </Tooltip>
  </div>
);
```
