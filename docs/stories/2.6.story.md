## Technical Notes

### OAuth PKCE Flow

**Step 1: Generate code_verifier + code_challenge (Frontend)**
```javascript
const codeVerifier = generateRandomString(128); // 128 chars
const codeChallenge = BASE64URL(SHA256(codeVerifier));

// Store in sessionStorage (client-side only)
sessionStorage.setItem('code_verifier', codeVerifier);
sessionStorage.setItem('code_challenge', codeChallenge);

// Redirect to OAuth provider
window.location = `https://slack.com/oauth/v2/authorize?client_id=XXX&code_challenge=${codeChallenge}&code_challenge_method=S256`;
```

**Step 2: Exchange code for token (Backend)**
```javascript
// User redirected back with authorization code
const code = req.query.code;
const codeVerifier = req.session.code_verifier; // From session

const response = await fetch('https://slack.com/api/oauth.v2.access', {
  method: 'POST',
  body: {
    client_id: process.env.SLACK_CLIENT_ID,
    // NO client_secret here - PKCE replaces it
    code: code,
    code_verifier: codeVerifier  // Prove we generated the challenge
  }
});

const token = response.access_token;

// Encrypt and store
const encryptedToken = encrypt(token, kmsKeyId);
await db.slackTokens.create({
  user_id: userId,
  workspace_id: response.team_id,
  encrypted_token: encryptedToken,
  kms_key_id: kmsKeyId,
  token_type: 'bot'
});
```

### Token Encryption (KMS)
```javascript
const AWS = require('aws-sdk');
const kms = new AWS.KMS();

// Encrypt
const encryptedToken = await kms.encrypt({
  KeyId: process.env.KMS_KEY_ID,
  Plaintext: plainToken
}).promise();

// Decrypt (when needed to make API call)
const decryptedToken = await kms.decrypt({
  CiphertextBlob: encryptedToken
}).promise();

// Key rotation: Track which key was used, support multiple keys
```

### Slack Message Template
```javascript
const template = "Rule {{rule_name}} executed in {{duration_ms}}ms - Status: {{status}}";
const variables = {
  rule_name: 'High Priority Task Alert',
  duration_ms: 245,
  status: 'success'
};

const message = template.replace(/\{\{(\w+)\}\}/g, (match, key) => variables[key]);
// Output: "Rule High Priority Task Alert executed in 245ms - Status: success"
```

### Rate Limiting
```javascript
// Redis counter: "{integration}:{workspace_id}:requests"
const key = `slack:${workspace_id}:requests`;
const count = await redis.incr(key);
if (count === 1) await redis.expire(key, 60); // Reset every minute

if (count > 60) {
  // Rate limited
  const retryAfter = await redis.ttl(key);
  throw new Error(`Rate limited. Retry after ${retryAfter}s`);
}
```

### Slash Command (Slack)
```javascript
// Webhook endpoint: POST /slack/slash-commands
app.post('/slack/slash-commands', async (req, res) => {
  const { command, text, user_id, team_id } = req.body;

  if (command === '/automation') {
    const ruleId = text.split(' ')[0];
    const result = await executeRule(ruleId, { source: 'slack_slash' });

    res.json({
      response_type: 'in_channel',
      text: `Rule execution started. Check dashboard for status.`,
      blocks: [
        { type: 'section', text: { type: 'mrkdwn', text: `*Status:* ${result.status}` } }
      ]
    });
  }
});
```
