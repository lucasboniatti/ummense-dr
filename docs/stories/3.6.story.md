## Technical Notes

### WebSocket Real-Time Architecture

```typescript
// Backend: WebSocket event broadcasting
import { Server } from 'socket.io';

const io = new Server(httpServer, {
  cors: { origin: process.env.FRONTEND_URL },
  reconnection: true,
  reconnectionDelay: 1000
});

// On execution completed
async function broadcastExecution(execution: Execution) {
  io.to(`user:${execution.user_id}`).emit('execution:completed', {
    id: execution.id,
    automationId: execution.automation_id,
    status: execution.status,
    duration: execution.duration_ms,
    timestamp: execution.completed_at,
    error: execution.error_context?.message
  });
}

// On DLQ update
async function broadcastDLQUpdate(dlqItem: DLQItem) {
  io.to(`user:${dlqItem.user_id}`).emit('dlq:updated', {
    id: dlqItem.id,
    automationId: dlqItem.automation_id,
    status: dlqItem.status,
    webhookUrl: dlqItem.webhook_url,
    timestamp: dlqItem.created_at
  });
}

// Frontend: React hook for real-time updates
function useDashboardRealtimeUpdates() {
  const [executions, setExecutions] = useState<Execution[]>([]);
  const [socket, setSocket] = useState<Socket | null>(null);

  useEffect(() => {
    const newSocket = io(process.env.REACT_APP_API_URL, {
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionDelayMax: 30000
    });

    newSocket.emit('dashboard:subscribe');

    newSocket.on('execution:completed', (execution: Execution) => {
      setExecutions(prev => [execution, ...prev].slice(0, 100));  // Keep last 100
    });

    newSocket.on('dlq:updated', (dlqItem: DLQItem) => {
      // Update DLQ widget
    });

    setSocket(newSocket);

    return () => newSocket.close();
  }, []);

  return { executions, socket };
}
```

### Analytics Materialized Views

```sql
-- Materialized view for daily metrics (refreshed hourly)
CREATE MATERIALIZED VIEW dashboard_daily_metrics AS
SELECT
  DATE(ae.created_at) as execution_date,
  COUNT(*) as total_executions,
  SUM(CASE WHEN ae.status = 'success' THEN 1 ELSE 0 END)::float / COUNT(*) * 100 as success_rate,
  AVG(ae.duration_ms) as avg_duration_ms,
  COUNT(CASE WHEN ae.status = 'failed' THEN 1 END) as failure_count
FROM automation_executions ae
GROUP BY DATE(ae.created_at)
ORDER BY execution_date DESC;

CREATE INDEX idx_dashboard_daily_metrics_date
ON dashboard_daily_metrics(execution_date);

-- Refresh job (hourly)
SELECT refresh_materialized_view('dashboard_daily_metrics');

-- Connector health view
CREATE MATERIALIZED VIEW connector_health AS
SELECT
  SUBSTRING(wd.webhook_id FROM 1 FOR 20) as connector_name,
  COUNT(*) as total_deliveries,
  SUM(CASE WHEN wd.status = 'success' THEN 1 ELSE 0 END)::float / COUNT(*) * 100 as success_rate,
  MAX(wd.created_at) as last_execution,
  COUNT(CASE WHEN wd.status = 'failed' THEN 1 END) as failure_count
FROM webhook_deliveries wd
WHERE wd.created_at > now() - INTERVAL '7 days'
GROUP BY connector_name
ORDER BY success_rate DESC;
```

### PDF Report Generation

```typescript
import { jsPDF } from 'jspdf';
import html2canvas from 'html2canvas';

async function generateReport(
  dashboardSnapshot: React.ReactNode,
  metrics: DashboardMetrics,
  dateRange: { start: Date; end: Date }
) {
  const pdf = new jsPDF();

  // Title and metadata
  pdf.setFontSize(20);
  pdf.text('Automation Execution Report', 20, 20);
  pdf.setFontSize(10);
  pdf.text(`Date: ${new Date().toLocaleDateString()}`, 20, 30);
  pdf.text(`Period: ${dateRange.start.toLocaleDateString()} - ${dateRange.end.toLocaleDateString()}`, 20, 36);

  // Convert React component to canvas
  const canvas = await html2canvas(dashboardSnapshot as HTMLElement);
  const imgData = canvas.toDataURL('image/png');
  pdf.addImage(imgData, 'PNG', 20, 50, 170, 100);

  // Metrics section
  pdf.addPage();
  pdf.setFontSize(14);
  pdf.text('Key Metrics', 20, 20);
  pdf.setFontSize(10);
  pdf.text(`Total Executions: ${metrics.totalExecutions}`, 20, 35);
  pdf.text(`Success Rate: ${metrics.successRate.toFixed(2)}%`, 20, 45);
  pdf.text(`Avg Duration: ${metrics.avgDuration.toFixed(0)}ms`, 20, 55);
  pdf.text(`Failures: ${metrics.failureCount}`, 20, 65);

  // Save
  pdf.save(`Synkra_Automation_Report_${new Date().toISOString().split('T')[0]}.pdf`);
}
```
